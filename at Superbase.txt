CREATE OR REPLACE PROCEDURE public.create_new_table(IN table_name TEXT, IN query TEXT)
LANGUAGE plpgsql
AS $$
BEGIN
    -- Construct the full query to create the table
    EXECUTE 'CREATE TABLE IF NOT EXISTS ' || table_name || ' (' || query || ')';
END;
$$;
